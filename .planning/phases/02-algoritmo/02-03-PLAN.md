## Input

```typescript
interface GenerationInput {
  month: number           // 1-12
  year: number
  employees: EmployeeWithAreas[]   // each has .areas[] with areaId + areaName
  templates: ShiftTemplateFlat[]   // each has .dayOfWeek, .areaId, .requiredCount, .isWildcard
  absences: AbsenceRecord[]        // each has .employeeId, .startDate, .endDate (YYYY-MM-DD)
}
```

## Phase 1 — Build helper data structures

```
daysInMonth = number of days in (year, month)
monthDates = array of all Date objects for each day: [2026-03-01, 2026-03-02, ...]

// Precompute absence lookup: Set<"employeeId-YYYY-MM-DD">
absentOnDay = new Set<string>()
for each absence in input.absences:
  for each date in absence.startDate..absence.endDate:
    absentOnDay.add(`${absence.employeeId}-${dateStr}`)

// Precompute area enablement: Map<employeeId, Set<areaId>>
employeeAreas = new Map<number, Set<number>>()
for each employee: employeeAreas.set(employee.id, new Set(employee.areas.map(a => a.areaId)))

// Zona crítica: days 28-31 of current month + days 1-2 of next month
// In practice: a date is "crítica" if it falls in days 28+ of the month AND is within 2 days of month-end,
// OR if it's the 1st or 2nd day of the following month (handled by checking day number >= 28)
// Simple rule: dayOfMonth >= 28 is zona crítica
isCritica = (date: Date) => date.getUTCDate() >= 28

// Area special day restrictions
MARKING_AREA_NAME = "Marking"          // only dayOfWeek 2 (Tue) and 4 (Thu)
VALERY_AREA_NAME = "Valery Camacho"    // only dayOfWeek 3 (Wed)
BUFFET_AREA_NAME = "Buffet"            // never dayOfWeek 7 (Sun)
```

## Phase 2 — Determine rest days per employee

```
For each employee:
  validRestDays = monthDates.filter(d =>
    !isCritica(d) &&
    getDayOfWeek(d) !== 6  // not Saturday (6 = Sat in 1=Mon..7=Sun)
  )
  sundays = validRestDays.filter(d => getDayOfWeek(d) === 7)

  // Pick 1 mandatory Sunday rest
  restSunday = sundays[Math.floor(Math.random() * sundays.length)]  // or rotate fairly

  // Pick 3 ordinary rest days from remaining valid non-Sunday days
  ordinaryPool = validRestDays.filter(d => d !== restSunday && getDayOfWeek(d) !== 7)
  restDays = [restSunday, ...pick3FromPool(ordinaryPool)]

  // GEN-13: if less than 3 ordinary days available, force remaining on next available
  if ordinaryPool.length < 3:
    forceDays from validRestDays (any valid day not already picked)
    log warning: "Employee X forced rest on day Y — insufficient valid days"

  employeeRestDays.set(employee.id, new Set(restDays.map(formatDate)))
```

## Phase 3 — Assign work shifts day by day

```
shifts: GeneratedShift[] = []
log: LogEntry[] = []

// Group templates by (dayOfWeek, areaId) for fast lookup
templatesByDayArea = groupBy(templates, t => `${t.dayOfWeek}-${t.areaId}`)

for each date in monthDates:
  dayOfWeek = getISODayOfWeek(date)  // 1=Mon..7=Sun
  dateStr = format(date, 'YYYY-MM-DD')

  // Determine which templates are active today
  todayTemplates = templates.filter(t =>
    t.dayOfWeek === dayOfWeek &&
    !t.isWildcard &&
    isAreaAllowedToday(t.areaName, dayOfWeek)  // Marking/Valery/Buffet rules
  )

  // isAreaAllowedToday(areaName, dow):
  //   if areaName === "Marking": return dow === 2 || dow === 4
  //   if areaName === "Valery Camacho": return dow === 3
  //   if areaName === "Buffet": return dow !== 7
  //   return true

  for each template in todayTemplates:
    needed = template.requiredCount
    assigned = 0

    // Get eligible employees for this template's area today
    eligible = employees.filter(emp =>
      !absentOnDay.has(`${emp.id}-${dateStr}`) &&
      employeeAreas.get(emp.id)?.has(template.areaId) &&
      !employeeRestDays.get(emp.id)?.has(dateStr)
    )

    // Apply rotation: morning vs afternoon preference
    isoWeek = getISOWeek(date)
    preferMorning = (isoWeek % 2 === 1)  // odd week = morning preference
    eligible = sortByRotationPreference(eligible, template.timeSlot, preferMorning)

    // Apply split shift limit: max 2 per employee per week
    isSplit = template.timeSlot.includes('|')
    if isSplit:
      eligible = eligible.filter(emp => splitShiftCountThisWeek(emp.id, isoWeek) < 2)

    // Assign up to requiredCount
    for emp in eligible[:needed]:
      shifts.push({ employeeId: emp.id, date: dateStr, timeSlot: template.timeSlot, areaId: template.areaId, isRest: false })
      assigned++
      if isSplit: incrementSplitCount(emp.id, isoWeek)

    if assigned < needed:
      log.push({ type: 'warning', message: `Area ${template.areaName} day ${dateStr}: needed ${needed}, assigned ${assigned}` })

  // Add rest day records for employees resting today
  for each employee with restDay === dateStr:
    if !absentOnDay.has(`${emp.id}-${dateStr}`):
      shifts.push({ employeeId: emp.id, date: dateStr, timeSlot: null, areaId: null, isRest: true })

  // Assign wildcard templates (isWildcard=true) for today's dayOfWeek
  wildcardTemplates = templates.filter(t => t.isWildcard && t.dayOfWeek === dayOfWeek)
  // Wildcard: assign 1 employee per wildcard template if available and not already assigned
  for each wildcardTemplate:
    alreadyAssignedToday = Set of employeeIds already in shifts for this date
    eligible = employees.filter(emp =>
      !alreadyAssignedToday.has(emp.id) &&
      !absentOnDay.has(`${emp.id}-${dateStr}`) &&
      employeeAreas.get(emp.id)?.has(wildcardTemplate.areaId)
    )
    if eligible.length > 0:
      emp = eligible[0]
      shifts.push({ employeeId: emp.id, date: dateStr, timeSlot: wildcardTemplate.timeSlot, areaId: wildcardTemplate.areaId, isRest: false })

return { shifts, log }
```

## Helper: Rotation sorting

```typescript
function sortByRotationPreference(employees, timeSlot, preferMorning) {
  const isMorningSlot = timeSlot.startsWith('7:') || timeSlot.startsWith('6:') || timeSlot.startsWith('8:')
  return employees.sort((a, b) => {
    if (preferMorning === isMorningSlot) return 0   // preferred match first
    return preferMorning ? -1 : 1
  })
}
```

## Note on fair distribution

The algorithm should distribute rest days and shift assignments as evenly as possible. Use a round-robin or rotation index per employee (modulo employee count) rather than always picking from the start of the sorted array. Track per-employee assignment counts and prefer employees with fewer assignments for equitable coverage.
