// Employee (table: employee)
model Employee {
  id: Int (autoincrement PK)
  name: String
  areas: EmployeeArea[]  // pivot
  shifts: Shift[]
  absences: Absence[]
}

// EmployeeArea (table: employee_area) — pivot
model EmployeeArea {
  employeeId: Int
  areaId: Int
  employee: Employee
  area: Area
  @@id([employeeId, areaId])
}

// Area (table: area)
model Area {
  id: Int (autoincrement PK)
  name: String (unique)  // "General", "Buffet", "Cosmetico", "Domicilio", "Electrodomestico", "Marking", "Valery Camacho"
  color: String
  shiftTemplates: ShiftTemplate[]
}

// ShiftTemplate (table: shift_template)
model ShiftTemplate {
  id: Int (autoincrement PK)
  areaId: Int
  dayOfWeek: Int        // 1=Mon, 2=Tue, 3=Wed, 4=Thu, 5=Fri, 6=Sat, 7=Sun
  timeSlot: String      // e.g. "7:00-14:00" or "7:00-11:00|11:30-14:30"
  isWildcard: Boolean   // comodín shift
  requiredCount: Int    // how many employees needed for this slot
  area: Area
}

// Shift (table: shift)
model Shift {
  id: Int (autoincrement PK)
  employeeId: Int
  date: DateTime @db.Date
  timeSlot: String?     // null means descanso (rest day)
  areaId: Int?          // null means descanso
  isManual: Boolean     // set when admin manually assigns
  @@unique([employeeId, date])  // one shift per employee per day
}

// Absence (table: absence)
model Absence {
  id: Int (autoincrement PK)
  employeeId: Int
  startDate: DateTime @db.Date
  endDate: DateTime @db.Date
  type: String          // "VAC" | "INC" | "PER" | "CAL" | "DESCANSO"
}

// GenerationLog (table: generation_log)
model GenerationLog {
  id: Int (autoincrement PK)
  month: Int
  year: Int
  createdAt: DateTime
  notes: Json           // array of LogEntry objects
}

    **lib/db/employees.ts:**
    ```typescript
    import { prisma } from "@/lib/db"
    import type { EmployeeWithAreas } from "@/types/generation"

    export async function getEmployeesWithAreas(): Promise<EmployeeWithAreas[]> {
      const employees = await prisma.employee.findMany({
        include: { areas: { include: { area: true } } },
        orderBy: { name: "asc" },
      })
      return employees.map(e => ({
        id: e.id,
        name: e.name,
        areas: e.areas.map(ea => ({ areaId: ea.areaId, areaName: ea.area.name })),
      }))
    }
    ```

    **lib/db/templates.ts:**
    ```typescript
    import { prisma } from "@/lib/db"
    import type { ShiftTemplateFlat } from "@/types/generation"

    export async function getShiftTemplates(): Promise<ShiftTemplateFlat[]> {
      const templates = await prisma.shiftTemplate.findMany({
        include: { area: true },
        orderBy: [{ areaId: "asc" }, { dayOfWeek: "asc" }],
      })
      return templates.map(t => ({
        id: t.id,
        areaId: t.areaId,
        areaName: t.area.name,
        dayOfWeek: t.dayOfWeek,
        timeSlot: t.timeSlot,
        isWildcard: t.isWildcard,
        requiredCount: t.requiredCount,
      }))
    }
    ```

    **lib/db/shifts.ts:**
    - `getShiftsForMonth(month, year)`: use `startOf/endOf` month via JS Date arithmetic — first day is `new Date(year, month-1, 1)`, last day is `new Date(year, month, 0)`. Prisma `gte`/`lte` filter on `date`.
    - `saveGeneratedShifts(shifts)`: use `prisma.$transaction()` with array of `prisma.shift.upsert()` per shift. On conflict (employeeId+date), skip if existing row has `isManual: true`. To implement skip-if-manual: first fetch existing manual shifts for the date range, then filter them out before upserting.
    - `clearShiftsForMonth(month, year)`: `prisma.shift.deleteMany({ where: { date: { gte: firstDay, lte: lastDay }, isManual: false } })`

    **lib/db/absences.ts:**
    - `getAbsencesForMonth(month, year)`: query with `startDate: { lte: lastDay }` AND `endDate: { gte: firstDay }` — overlap condition.

    **lib/db/generationLog.ts:**
    - `saveGenerationLog(month, year, log)`: `prisma.generationLog.upsert({ where: { month_year: { month, year } }, ... })`. Note: the GenerationLog model has no unique constraint on (month, year) by default — use `findFirst` + upsert pattern: delete existing if found, then create new.
    - `getGenerationLog(month, year)`: `prisma.generationLog.findFirst({ where: { month, year }, orderBy: { createdAt: "desc" } })`

    All date boundary calculations: use `new Date(Date.UTC(year, month-1, 1))` and `new Date(Date.UTC(year, month, 0))` for UTC-safe boundaries (Prisma Date type stores as UTC midnight).
